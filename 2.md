My Solution to [LeetCode problem 1. Two Sum](https://leetcode.com/submissions/detail/357498513/):

```
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        # create dict of nums and their indexes
        search_sub = {}
        
        for index, value in enumerate(nums):
            search_sub.update({value: index})
        
        # traverse through nums
        for i, el in enumerate(nums):
            # subtract each el w/ target
            sub = target - el

            # find that value in hash table
            try:
                # you may not use the same element twice
                if search_sub[sub] != i:
                    # return the index and el index
                    return [search_sub[sub], i]
            except KeyError:
                continue
        
        # If it didn't find a sum to target
        return None
```
