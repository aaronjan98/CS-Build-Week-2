        l1_p = l1

        l2_p = l2

                # tail = tail.next
                print('IF:','new_ll:', new_ll, 'tail:', tail)
            else:
                print('ELSE:', new_ll)
                current.next = ListNode(pointer.val)
                # tail = tail.next
        
        # loop through both linked lists entirely
        while l1_p and l2_p:
            
            # check if first value in l1 is less than in l2
            if (l2_p == None and l1 != None) or (l1_p.val <= l2_p.val):
                # call the new_ll
                # create_ll_node(l1_p)
                # increment l1 pointer
                if new_ll == None:

                    current = new_ll
